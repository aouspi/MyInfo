class LinkedLIst{

   Node head;
   
   Class Node{
      int data;
      Node next;
      
      Node(int data){
        this.data = data;
        this.next = null;
      }
   }

   public static LinkedList insert(Linked list, int data){
    
      Node new_node = new Node(data);
      new_node.next = null;

      if(list.head ==null){
         list.head = new_node;
      }
      else{
         Node last = list.head;
         while(last.next!=null){
            last = last.next;
         } 
         last.next = new_node;
      }
   }

}
